Sub Admin_Permissions()
'Unprotect Workbook with a Password Added by Viraj Lakshan Rathnayake on 20.03.2019
    ThisWorkbook.Unprotect Password:="Admin@donandsons.lk54321"
    
    For Each ws In ThisWorkbook.Worksheets
        ws.Visible = xlSheetVisible
        ws.Unprotect Password:="Admin@donandsons.lk54321"
    Next
    
    ActiveWindow.DisplayWorkbookTabs = True
    ActiveWindow.DisplayHorizontalScrollBar = True
    
    Sheet1.HomeButton.Visible = True
    
    Sheet1.MorningCurfewButton.Visible = False
    Sheet1.MorningApNewYearButton.Visible = False
    Sheet1.MorningPoyaButton.Visible = False
    Sheet1.MorningSaturdayButton.Visible = False
    Sheet1.MorningSundayButton.Visible = False
    Sheet1.MorningWeekdaysButton.Visible = False
    
    'Navigate to Dashboard
    Sheet27.Select
    
'Sheet21 LoginButton Visible False
    Sheet21.LoginButton.Visible = False
End Sub
