Sub Workbook_Open()


'Find This Workbook Name
Sheet2.Range("Y1").Value = Dir(ThisWorkbook.FullName)


Sheet21.TextBoxWelcomeGreeting.Value = ""

Dim GreetingText As String

 Select Case Time
 Case Is < TimeValue("12:00")
 GreetingText = "*Good Morning"
 
 Case Is < TimeValue("16:00")
 GreetingText = "*Good Afternoon"
 
 Case Else
 GreetingText = "*Good Evening"

End Select
 
 Sheet21.TextBoxWelcomeGreeting.Value = GreetingText & " " & Environ("username") & "*"

Sheet1.TextBoxGenarated.Visible = False

'Display False Page Breaks of Sheet1 ("Order")
    Sheet1.DisplayPageBreaks = False
    
'Display False Page Breaks of Sheet21 ("Home")
    Sheet21.DisplayPageBreaks = False

'Open as Sheet21 ("Home")
    Sheet21.Select
End Sub
