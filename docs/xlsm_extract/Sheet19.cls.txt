Private Sub Worksheet_Calculate()
Dim Username As String, Rng As Range
Username = Sheet21.Range("U1").Text

Set Rng = Worksheets("Users").Range("B2:F7")

Dim UsernameCheck As String

On Error Resume Next
UsernameCheck = Application.WorksheetFunction.VLookup(Username, Rng, 5, False)
On Error GoTo 0

    If UsernameCheck = "Admin" Then

        Sheet19.Range("B4").Value = Format(Now(), "dddd, mmmm dd, yyyy")

    End If
End Sub
