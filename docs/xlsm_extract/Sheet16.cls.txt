Private Sub Worksheet_Change(ByVal Target As Range)
Dim cell As Range

    With Intersect(ActiveSheet.Columns("B"), ActiveSheet.UsedRange)
        For Each cell In .Cells
            If WorksheetFunction.CountIf(.Resize(cell.Row - .Rows(1).Row + 1), cell.Value) > 1 Then MsgBox "              ***Duplicate Username Found***" & vbNewLine & vbNewLine & _
            "                         '" & cell.Value & "' in Cell " & cell.Address & vbNewLine & vbNewLine & "Duplicate Values has been Automatically Deleted ", vbCritical + vbOKOnly, "***EPF Payroll Programme***"
        Next cell
    End With

ActiveSheet.ListObjects("Table1").Range.RemoveDuplicates Columns:=1, Header:=xlYes

End Sub
