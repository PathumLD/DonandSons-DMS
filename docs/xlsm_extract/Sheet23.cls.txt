Private Sub EveningApNewYearPrintButton_Click()

Dim Username As String, Rng As Range

Set Rng = Worksheets("Users").Range("B2:F7")

On Error GoTo 0

    Sheet23.Unprotect Password:="Admin@donandsons.lk54321"

    Sheet23.Range("A1:CH77").AutoFilter Field:=86, Criteria1:="Y"

    Sheet23.PageSetup.PaperSize = xlPaperA4
    Sheet23.PageSetup.Orientation = xlLandscape
    Sheet23.PageSetup.FitToPagesTall = 1
    Sheet23.PageSetup.FitToPagesWide = 1
    Sheet23.PageSetup.PrintArea = "$CI$2:$DA$77"

    Sheet23.PageSetup.CenterFooter = "&""Cambria,Italic""&B&10This File Was Prepared by " & _
    Username & " on &D at &T"
    
    Application.Dialogs(xlDialogPrint).show
    
    Sheet23.ShowAllData
    Cells.AutoFilter
    
If UsernameCheck = "Admin" Then
    
    Sheet23.Unprotect Password:="Admin@donandsons.lk54321"
    
End If

End Sub

Private Sub Worksheet_Activate()

Dim Username As String, Rng As Range
Username = Sheet21.Range("U1").Text

Set Rng = Worksheets("Users").Range("B2:F7")

Dim UsernameCheck As String

On Error Resume Next
UsernameCheck = Application.WorksheetFunction.VLookup(Username, Rng, 5, False)
On Error GoTo 0

If UsernameCheck = "Admin" Then

    If Username = "Dulan" Or Username = "Hansa" Then
    
        Call Disable_Insert
    
    End If
    
ElseIf UsernameCheck = "Manager" Then

    Sheet23.Unprotect Password:="Admin@donandsons.lk54321"
 
End If

End Sub
